مثال عملي: برنامج حاسبة الدرجات

برنامج بسيط لحساب معدل درجات الطالب:

// برنامج حساب معدل الدرجات

using

System;

class

Program {

static void

Main() {

// إعلان المتغيرات

string

studentName = "أحمد";

int

mathScore = 85;

int

scienceScore = 92;

int

englishScore = 78;

// حساب المجموع والمعدل

int

totalScore = mathScore + scienceScore + englishScore;

double

average = totalScore / 3.0;

// تحديد حالة النجاح

bool

isPassed = average >= 60;

// عرض النتائج

Console.WriteLine($"اسم الطالب: {studentName}"); Console.WriteLine($"المجموع: {totalScore}"); Console.WriteLine($"المعدل: {average:F2}"); Console.WriteLine($"حالة النجاح: {isPassed}"); } }

مخرجات البرنامج:

اسم الطالب: أحمد
المجموع: 255
المعدل: 85.00
حالة النجاح: True

المفاهيم المستخدمة:

أنواع المتغيرات

string, int, double, bool

العمليات الحسابية

الجمع والقسمة

المقارنات المنطقية

أكبر من أو يساوي (>=)

طباعة النصوص

استخدام الـ String Interpolation

تحدي للطلاب:

قم بتعديل البرنامج لإضافة:

  1. درجة مادة رابعة (الحاسوب)
  2. تقدير الطالب (ممتاز، جيد جداً، جيد، مقبول، راسب)

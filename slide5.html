قواعد تسمية المتغيرات:

  • يجب أن تبدأ بحرف أو _ (underscore)
  • لا يمكن استخدام مسافات أو رموز خاصة
  • لا يمكن استخدام الكلمات المحجوزة
  • يفضل استخدام أسماء ذات معنى

طرق إعلان المتغيرات:

// الإعلان فقط (بدون تهيئة)

int

score;

string

playerName;

// الإعلان مع التهيئة

int

score = 100;

string

playerName = "لاعب1";

// استخدام var (استنتاج النوع تلقائياً)

var

age = 25; // سيكون من نوع int

var

name = "أحمد"; // سيكون من نوع string
